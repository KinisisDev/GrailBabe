# Redis tier & whether it's reachable
az redis show -g $RG -n redis-grailbabedev-dev \
  --query "{sku:sku.name, capacity:sku.capacity, host:hostName, state:provisioningState}" -o table

# Service Bus tier
az servicebus namespace show -g $RG -n sb-grailbabedev-dev \
  --query "{sku:sku.name, tier:sku.tier, status:status}" -o table

# Service Bus queues/topics — empty = definitely unused
az servicebus queue list -g $RG --namespace-name sb-grailbabedev-dev --query "[].name" -o tsv
az servicebus topic list -g $RG --namespace-name sb-grailbabedev-dev --query "[].name" -o tsv

# The three svc container apps — image + last revision
for app in ca-grailbabedev-svc1-dev ca-grailbabedev-svc2-dev ca-grailbabedev-svc3-dev; do
  echo "=== $app ==="
  az containerapp show -g $RG -n $app \
    --query "{image:properties.template.containers[0].image, replicas:properties.template.scale, status:properties.runningStatus}" -o json
done
